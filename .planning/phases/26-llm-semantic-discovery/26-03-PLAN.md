```go
func TestLLMSemanticRelationships_RealCorpus(t *testing.T) {
    // Load real corpus.
    corpusDir := filepath.Join("..", "..", "test-data", "graph-test-docs")
    if _, err := os.Stat(corpusDir); err != nil {
        t.Skipf("graph-test-docs not found at %s: %v", corpusDir, err)
    }

    scanner := NewScanner(ScanConfig{Dir: corpusDir})
    docs, err := scanner.Scan()
    if err != nil || len(docs) == 0 {
        t.Skipf("scan failed or empty: %v", err)
    }

    // Subtest 1: empty trees — must return nil without panic.
    t.Run("EmptyTrees", func(t *testing.T) {
        cfg := DefaultLLMDiscoveryConfig()
        result := LLMSemanticRelationships(docs, []FileTree{}, []string{"auth", "db"}, cfg)
        if result != nil {
            t.Errorf("expected nil for empty trees, got %d edges", len(result))
        }
    })

    // Subtest 2: stub binary — graceful degradation.
    t.Run("StubBinary", func(t *testing.T) {
        tmpDir := t.TempDir()
        // Build minimal trees from docs (just file paths, no summaries).
        trees := make([]FileTree, len(docs))
        for i, doc := range docs {
            trees[i] = FileTree{
                File: doc.RelPath,
                Root: &TreeNode{Heading: doc.RelPath, Summary: "stub summary"},
            }
        }
        components := []string{"auth-service", "api-gateway", "user-service"}

        cfg := DefaultLLMDiscoveryConfig()
        cfg.ExecutablePath = "/nonexistent/pageindex" // Will fail, return empty
        cfg.CacheFile = filepath.Join(tmpDir, "cache.json")

        result := LLMSemanticRelationships(docs, trees, components, cfg)
        // With stub binary, pageindex fails gracefully — expect 0 or nil edges.
        t.Logf("RealCorpus stub: %d docs, %d trees, %d edges returned", len(docs), len(trees), len(result))
        // No panic is the primary assertion.
    })
}
```

```go
func TestLLMDiscoveryCache_RoundTrip(t *testing.T) {
    tmpDir := t.TempDir()
    cacheFile := filepath.Join(tmpDir, "roundtrip.json")

    // Create cache with 2 entries.
    original := &llmDiscoveryCache{
        Entries: []llmDiscoveryCacheEntry{
            {
                DocID:       "api.md",
                ContentHash: hashDocContent("# API Service"),
                Results: []llmDiscoveryResult{
                    {Target: "auth", Relationship: "depends-on", Confidence: 0.8, Evidence: `"auth token"`},
                },
                GeneratedAt: time.Now().Truncate(time.Second), // truncate for JSON round-trip
            },
            {
                DocID:       "worker.md",
                ContentHash: hashDocContent("# Worker Service"),
                Results: []llmDiscoveryResult{
                    {Target: "db", Relationship: "calls", Confidence: 0.75, Evidence: `"database query"`},
                },
                GeneratedAt: time.Now().Truncate(time.Second),
            },
        },
        GeneratedAt: time.Now().Truncate(time.Second),
    }

    if err := saveLLMDiscoveryCache(cacheFile, original); err != nil {
        t.Fatalf("save failed: %v", err)
    }

    loaded := loadLLMDiscoveryCache(cacheFile)
    if loaded == nil {
        t.Fatal("load returned nil")
    }
    if len(loaded.Entries) != 2 {
        t.Errorf("round-trip: expected 2 entries, got %d", len(loaded.Entries))
    }

    // Verify edges are produced from loaded cache.
    edges := buildEdgesFromCache(loaded, []string{"auth", "db", "cache"})
    if len(edges) != 2 {
        t.Errorf("round-trip edges: expected 2, got %d", len(edges))
    }
}
```

```go
}
```
